  customer_id, total_deals, last_payment_date

SELECT 
  c.id, COUNT(d.id), MAX(p.date)
FROM customers c
JOIN deals d ON d.customer_id = c.id
LEFT JOIN payments p ON p.deal_id = d.id
GROUP BY c.id;
